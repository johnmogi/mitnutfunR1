                            console.error('Invalid date in formatDate:', date);
                            return '';
                        }
                        
                        const day = String(date.getDate()).padStart(2, '0');
                        const month = String(date.getMonth() + 1).padStart(2, '0');
                        const year = date.getFullYear();
                        
                        return `${day}.${month}.${year}`;

                        if (!dateRange || typeof dateRange !== 'string' || !dateRange.includes(' - ')) {
                            return [];

                        const dates = [];
                        
                        try {
                            const startParts = startDate.split('.');
                            const endParts = endDate.split('.');
                            
                            let currentDate = new Date(`${startParts[2]}-${startParts[1]}-${startParts[0]}`);
                            const end = new Date(`${endParts[2]}-${endParts[1]}-${endParts[0]}`);
                            
                            if (isNaN(currentDate.getTime()) || isNaN(end.getTime())) {
                                console.error('Invalid date range:', { startDate, endDate });
                                return [];
                            }
                            
                            while (currentDate <= end) {
                                dates.push(formatDate(currentDate));
                                currentDate.setDate(currentDate.getDate() + 1);
                            }
                        } catch (error) {
                            console.error('Error processing date range:', error);
                        }
                        
                        return dates;
